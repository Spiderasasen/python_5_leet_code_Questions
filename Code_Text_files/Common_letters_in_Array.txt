class Solution:
    def longestCommonPrefix(self, strs: list[str]) -> str:
        final_strings = ""
        if len(strs) == 1:
            return strs[0]
        for i in range(len(strs[0])):
            char = strs[0][i] #character that will check though all other words
            for strn in strs[1:]:  # Hint: Use a slice to skip the first word
                if i == len(strn) or strn[i] != char:
                    return final_strings
            final_strings += char
        return final_strings
